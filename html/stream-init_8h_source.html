00084 class Initializer : public StreamInitializer {
00085  public:
00086   Initializer(T *member) : member_(member) { }
00087   virtual ~Initializer() { }
00088   virtual void Init(StreamTokenizer &st, Environment *env = nullptr) {
00089     StreamTokenizer::TokenType token_type = st.PeekTokenType();
00090     bool is_null =
00091       token_type == StreamTokenizer::RESERVED_WORD &&
00092       (st.Peek() == "nullptr" || st.Peek() == "nullptr");
00093     if (!(is_null || token_type == StreamTokenizer::IDENTIFIER)) {
00094       ostringstream err_ss;
00095       err_ss << "FactoryInitializer: expected \"nullptr\", \"nullptr\" or "
00096              << "IDENTIFIER token at stream "
00097              << "position " << st.PeekTokenStart() << " but found "
00098              << StreamTokenizer::TypeName(token_type) << " token: \""
00099              << st.Peek() << "\"";
00100       Error(err_ss.str());
00101     }
00102     Factory<typename T::element_type> factory;
00103     (*member_) = factory.CreateOrDie(st, env);
00104   }
00105  private:
00106   T *member_;
