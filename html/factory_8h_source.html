00762   shared_ptr<T> CreateOrDie(StreamTokenizer &st, Environment *env = nullptr) {
00763     shared_ptr<Environment> env_ptr(env == nullptr ?
00764                                     Environment::CreateEmpty() : env->Copy());
00765     size_t start = st.PeekTokenStart();
00766     StreamTokenizer::TokenType token_type = st.PeekTokenType();
00767     if (token_type == StreamTokenizer::RESERVED_WORD &&
00768         (st.Peek() == "nullptr" || st.Peek() == "NULL")) {
00769       // Consume the nullptr.
00770       st.Next();
00771       return shared_ptr<T>();
00772     }
00773     if (token_type != StreamTokenizer::IDENTIFIER) {
00774       ostringstream err_ss;
00775       err_ss << "Factory<" << BaseName() << ">: "
00776              << "error: expected type specifier token but found "
00777              << StreamTokenizer::TypeName(token_type);
00778       Error(err_ss.str());
00779     }
00780 
00781     // Read the concrete type of object to be created.
00782     string type = st.Next();
00783 
00784     // Read the open parenthesis token.
00785     if (st.Peek() != "(") {
00786       ostringstream err_ss;
00787       err_ss << "Factory<" << BaseName() << ">: "
00788              << "error: expected '(' at stream position "
00789              << st.PeekTokenStart() << " but found \"" << st.Peek() << "\"";
00790       Error(err_ss.str());
00791     }
00792     st.Next();
00793 
00794     // Attempt to create an instance of type.
00795     typename unordered_map<string, const Constructor<T> *>::iterator cons_it =
00796         cons_table_->find(type);
00797     if (cons_it == cons_table_->end()) {
00798       ostringstream err_ss;
00799       err_ss << "Factory<" << BaseName() << ">: "
00800              << "error: unknown type: \"" << type << "\"";
00801       Error(err_ss.str());
00802     }
00803     shared_ptr<T> instance(cons_it->second->NewInstance());
00804 
00805     // Ask new instance to set up member initializers.
00806     Initializers initializers;
00807     instance->RegisterInitializers(initializers);
00808 
00809     // Parse initializer list.
00810     while (st.Peek() != ")") {
00811       token_type = st.PeekTokenType();
00812       if (token_type != StreamTokenizer::IDENTIFIER) {
00813         ostringstream err_ss;
00814         err_ss << "Factory<" << BaseName() << ">: "
00815                << "error: expected token of type IDENTIFIER at "
00816                << "stream position " << st.PeekTokenStart() << " but found "
00817                << StreamTokenizer::TypeName(token_type) << ": \""
00818                << st.Peek() << "\"";
00819         Error(err_ss.str());
00820       }
00821       size_t member_name_start = st.PeekTokenStart();
00822       string member_name = st.Next();
00823       typename Initializers::iterator init_it = initializers.find(member_name);
00824       if (init_it == initializers.end()) {
00825         ostringstream err_ss;
00826         err_ss << "Factory<" << BaseName() << ">: "
00827                << "error: unknown member name \"" << member_name
00828                << "\" in initializer list for type " << type << " at stream "
00829                << "position " << member_name_start;
00830         Error(err_ss.str());
00831       }
00832       MemberInitializer *member_initializer = init_it->second;
00833 
00834       // Read open parenthesis.
00835       if (st.Peek() != "(") {
00836         ostringstream err_ss;
00837         err_ss << "Factory<" << BaseName() << ">: "
00838                << "error: expected '(' at stream position "
00839                << st.PeekTokenStart() << " but found \"" << st.Peek() << "\"";
00840         Error(err_ss.str());
00841       }
00842       st.Next();
00843 
00844       // Initialize member based on following token(s).
00845       member_initializer->Init(st, env_ptr.get());
00846 
00847       // Read close parenthesis for current member initializer.
00848       if (st.Peek() != ")") {
00849         ostringstream err_ss;
00850         err_ss << "Factory<" << BaseName() << ">: "
00851                << "error: expected ')' at stream position "
00852                << st.PeekTokenStart() << " but found \"" << st.Peek() << "\"";
00853         Error(err_ss.str());
00854       }
00855       st.Next();
00856 
00857       // Each member initializer must be followed by a comma or the final
00858       // closing parenthesis.
00859       if (st.Peek() != ","  && st.Peek() != ")") {
00860         ostringstream err_ss;
00861         err_ss << "Factory<" << BaseName() << ">: "
00862                << "error: expected ',' or ')' at stream position "
00863                << st.PeekTokenStart() << " but found \"" << st.Peek() << "\"";
00864         Error(err_ss.str());
00865       }
00866       // Read comma, if present.
00867       if (st.Peek() == ",") {
00868         st.Next();
00869       }
00870     }
00871 
00872     // Read the close parenthesis token for this factory type specification.
00873     if (st.Peek() != ")") {
00874       ostringstream err_ss;
00875       err_ss << "Factory<" << BaseName() << ">: "
00876              << "error: expected ')' at stream position "
00877              << st.PeekTokenStart() << " but found \"" << st.Peek() << "\"";
00878       Error(err_ss.str());
00879     }
00880     st.Next();
00881 
00882     // Run through all member initializers: if any are required but haven't
00883     // been invoked, it is an error.
00884     for (typename Initializers::const_iterator init_it = initializers.begin();
00885          init_it != initializers.end();
00886          ++init_it) {
00887       MemberInitializer *member_initializer = init_it->second;
00888       if (member_initializer->Required() &&
00889           member_initializer->Initialized() == 0) {
00890         ostringstream err_ss;
00891         err_ss << "Factory<" << BaseName() << ">: "
00892                << "error: initialization for member with name \""
00893                << init_it->first << "\" required but not found (current "
00894                << "stream position: " << st.tellg() << ")";
00895         Error(err_ss.str());
00896       }
00897     }
00898 
00899     size_t end = st.tellg();
00900     // Invoke new instance's Init method.
00901     string stream_str = st.str();
00902     //cerr << "Full stream string is: \"" << stream_str << "\"" << endl;
00903     string init_str = stream_str.substr(start, end - start);
00904     //cerr << "PostInit string is: \"" << init_str << "\"" << endl;
00905     instance->PostInit(env_ptr.get(), init_str);
00906 
00907     return instance;
00908   }
